#include <iostream>
using namespace std;
class Array{
    private:
        int *array;
        int *t;
        int length;
        int capacity;
    public:
        Array(int size){
            this -> capacity = size;
            this -> length = 0;
            array = new int[capacity];
            t = nullptr;
        }
        
        bool isEmpty(){
            if(length == 0){
                return true;
            }  
            return false;
        }
        
        bool isFull(){
            if(length == capacity){
                return true;
            }
            return false;
        }
        
        void insertData(){
            if(isFull()){
                return;
            }
            
            int number = 0;
            if(isEmpty()){
                cout << "Enter number to insert: " << endl;
                cin >> number;
                
                t = array;
                *t = number;
                length++ ;
                return;
            }
            
            int position;
            cout << "Enter position to insert: " ;
            cin >> position;
            
            if(position >= 1 && position <= (length + 1)){
                // case 1:
                int number;
                cout << "Enter number: " << endl;
                cin >> number;
                
                // position - enter at last just like the (push) function of an array.
               if(position == length + 1){
                    t = array;
                    *(t + length) = number;
                    length++ ;
                    return ;
                }
                
                // case 2: enter in the middle
                for(int i = length - 1; i >= position - 1; i--){
                     t = array;
                    *(t + i + 1) = *(t + i);
                }
                    *(t+position-1) = number;
                    length++ ;
            }
            else{
                cout << "Invalid Position. Enter a valid position." << endl;
            }
            
        }
        
        void display(){
            // use t - helper pointer to traverse..
            t = array;
            
            cout << "Array Elements: " << endl;
            for (int i=0 ; i<length ; i++){
                cout << *(t + i) << " " ;
            }
        }
        
        ~Array(){
            delete[] array;
        }
};

int main(){
    // array..
        // length - full
        // length - empty
        // insert data. 
            // insert at end.
            // insert at start.
            // insert at middle.
        // display data.
        
    Array a1(4);
    a1.insertData();
    a1.insertData();
    a1.insertData();
    
    a1.display();
    
    return 0;
}
